Inspect the URL for this page and note the text after the question mark. The question mark and the text that follow is a "query string" sent from a form using method="GET".

?firstName=Fred&lastName=Jones

The first value in the query string starts with a "?" followed by a name and a value.

?firstName=Fred

The second and all remaining values in the query string starts with an "&" followed by a name and a value.

&lastName=Jones

<h1>Your Submitted Data</h1>
<div id="yourdata"></div>
<script>
// get the div to display results
var yourdataDIV = document.getElementById("yourdata");
// get the URL for the current page
var url = window.location.href;
// pull off the query string (everything after the "?")
var urlparts = url.split("?");
if (urlparts.length == 0)
{
  yourdataDIV.innerHTML = "No query string found";
}
else
{
  // pull apart the name/value pairs
  dataparts = urlparts[1].split("&")
  // display them
  for (var i=0; i<dataparts.length; i++)
  {
    yourdataDIV.innerHTML += dataparts[i] + "<br>";
  }
}
</script>
